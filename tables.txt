CREATE DATABASE food_app;
USE food_app;

CREATE TABLE users (
    id INTEGER NOT NULL AUTO_INCREMENT,
    username text NOT NULL,
    password text NOT NULL,
    name text NOT NULL,
    email text NOT NULL,
    avatar text NOT NULL,
    PRIMARY KEY (id)
);

CREATE TABLE photos (
    id INTEGER NOT NULL AUTO_INCREMENT,
    filename text NOT NULL,
    owner  INTEGER NOT NULL,
    timestamp Timestamp,
    caption VARCHAR(1000),
    PRIMARY KEY (id)
);
